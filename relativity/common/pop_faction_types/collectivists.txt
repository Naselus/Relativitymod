collectivist_faction = {
	lose_support_on_pop_leave = no

	is_potential = {
		always = yes
	}
	
	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}
	mandate = {
		key = "mandate_emancipation2"
		parameter_transfer = {}
		weight = {
			base = 20
		}
	}

	mandate = {
		key = "mandate_miner"
		parameter_transfer = {}
		weight = {
			base = 20
		}
	}

	mandate = {
		key = "mandate_orbital_researcher"
		parameter_transfer = {}
		weight = {
			base = 20
		}
	}

	mandate = {
		key = "mandate_slaver"
		parameter_transfer = {}
		weight = {
			base = 20
		}
	}
	can_create = {
		always = yes
	}
	
	flag = {
		icon = {
			category = "special"
			file = "ethic_collectivist.dds"
		}
		background= {
			category = "backgrounds"
			file = "new_dawn.dds"
		}
		colors={
			"dark_teal"
			"teal"
			"null"
			"null"
		}
	}
	
	
	on_create = {
		set_name = random
		set_support = 1.0
		owner = {
			if = {
				limit = {
					any_owned_leader = {
						leader_class = ruler
						OR = {
							has_trait = leader_fcollectivist 
							has_trait = leader_collectivist 
						}					
					}
				}
				random_owned_leader = {
					limit = { 
						leader_class = ruler
						OR = {
							has_trait = leader_fcollectivist 
							has_trait = leader_collectivist 
						}
					}
					root = {
						assign_leader = prev
					}
				}
				else = {
					if = {
						limit = {
							any_owned_leader = {
								leader_of_faction = no 
								OR = {
									has_trait = leader_fcollectivist 
									has_trait = leader_collectivist 
								}					
							}
						}
						random_owned_leader = {
							limit = { 
								leader_of_faction = no 
								OR = {
									has_trait = leader_fcollectivist 
									has_trait = leader_collectivist 
								}
							}
							root = {
								assign_leader = prev
							}
						}
						else = {
							create_leader = {
								type = governor
								species = prevprev
								name = random
								skill = 0
								traits = {
								has_trait = leader_fcollectivist
								add_random_leader_trait = yes
								}
							}
							last_created_leader = {
								recruitable = no
								root = {
									assign_leader = prev
								}
							}					
						}
					}
				}
			}
			every_owned_pop = {
				if = {
					limit = { 
						member_of_faction = no
						OR = {
							pop_has_ethic = ethic_fanatic_collectivist
						}
					}
					set_pop_faction = ROOT
				}
			}
		}
	}

			}
		}
