		set_global_flag = asteroid_sighted_prim_global

# Incoming Asteroid
country_event = {
	id = observation.2
	title = "observation.2.name"
	desc = "observation.2.desc"
	picture = GFX_evt_asteroid_approaching_planet
	show_sound = event_red_alert
	location = FROM

	is_triggered_only = yes
	
	immediate = {
		create_country = {
			name = "Incoming Asteroid"
			type = faction
		}
		last_created_country = {
			create_fleet = { name = "Asteroid" }
			set_country_flag = incoming_asteroid_prim
			last_created_fleet = {
				set_owner = PREV
				set_fleet_flag = primitive_asteroid
				create_ship = {
					name = "Asteroid"
					design = "Asteroid"
				}
				set_location = {
					target = FROM
					distance = 75
					angle = random 
				}
				queue_actions = {
					move_to = FROM
					effect = {
						id = "observation.2.effect.1"
						ROOT = {
							random_list = {
								75 = { country_event = { id = observation.3 } }
								25 = { country_event = { id = observation.4 } }
							}
						}
						destroy_fleet = this
					}
				}
			}
		}
	}
	
	option = {
		name = "observation.2.a"
	}
}
