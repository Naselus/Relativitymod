# New Admiral
country_event = {
	id = leader.1
	title = "leader.1.name"
	desc = "leader.1.desc"
	picture = GFX_evt_federation_fleet
	location = ROOT
	
	is_triggered_only = yes
	
	trigger = {
		is_country_type = default
		free_leader_slots > 0
	}
	
	immediate = {
		FROMFROM = {
			solar_system = { save_event_target_as = system }
		}
		random_owned_ship = { 
			limit = {
				fleet = { is_same_value = FROMFROM }
			}
			save_event_target_as = ship
		}
		random_list = {
			25 = {
				create_leader = {
					type = admiral
					species = owner_main_species
					name = random
					skill = 201
					traits = {
						trait = leader_trait_gale_speed
						trait = leader_trait_aggressive
					}
				}
				last_created_leader = { save_event_target_as = officer }
			}
			25 = {
				create_leader = {
					type = admiral
					species = owner_main_species
					name = random
					skill = 301
					traits = {
						trait = leader_trait_aggressive
						trait = leader_trait_engineer
					}
				}
				last_created_leader = { save_event_target_as = officer }
			}
			25 = {
				create_leader = {
					type = admiral
					species = owner_main_species
					name = random
					skill = 401
					traits = {
						trait = leader_trait_fleet_logistician
						trait = leader_trait_cautious
					}
				}
				last_created_leader = { save_event_target_as = officer }
			}
			25 = {
				create_leader = {
					type = admiral
					species = owner_main_species
					name = random
					skill = 250
					traits = {
						trait = leader_trait_scout
						trait = leader_trait_trickster

				}
				last_created_leader = { save_event_target_as = officer }
			}
		}
	}
	
	option = {
		name = GOOD
		custom_tooltip = new_admiral
	}
}
